```typescript
// Helper status is a discriminated union; the status bar reads this verbatim.
export const HelperStatusSchema = z.discriminatedUnion('state', [
  z.object({ state: z.literal('starting') }),
  z.object({ state: z.literal('healthy'), pid: z.number(), version: z.string() }),
  z.object({ state: z.literal('restarting'), attempt: z.number(), nextInMs: z.number() }),
  z.object({ state: z.literal('offline'), reason: z.string(), since: z.number() }),
  z.object({ state: z.literal('crashed'), restartCount: z.number(), reason: z.string() }),
]);
export type HelperStatus = z.infer<typeof HelperStatusSchema>;

// IPC channel: 'app:bootstrap' — called once on renderer startup
export const AppBootstrapArgsSchema = z.void();
export const AppBootstrapResultSchema = z.object({
  firstRun: z.boolean(),
  userDataPath: z.string(),
  dataDir: z.string(),
  theme: z.enum(['system', 'dark', 'light']),
  helper: HelperStatusSchema,
  jdkFound: z.boolean(),
  jdkPath: z.string().nullable(),
});

// IPC channel: 'app:setDataDir'
export const SetDataDirArgsSchema = z.object({ path: z.string().min(1) });
export const SetDataDirResultSchema = z.object({ ok: z.boolean(), cloudSync: z.enum(['dropbox', 'onedrive', 'icloud', 'googledrive']).nullable() });

// IPC channel: 'app:showOpenDialog'
export const ShowOpenDialogArgsSchema = z.object({ kind: z.enum(['folder', 'file']), title: z.string().optional() });
export const ShowOpenDialogResultSchema = z.object({ path: z.string().nullable() });

// IPC channel: 'helper:getStatus' / 'helper:restart'
export const HelperGetStatusResultSchema = HelperStatusSchema;
export const HelperRestartResultSchema = HelperStatusSchema;

// IPC channel: 'request:diagnose' — Walking Skeleton probe
export const RequestDiagnoseArgsSchema = z.void();
export const RequestDiagnoseResultSchema = z.object({
  ok: z.boolean(),
  error: z.object({ code: z.string(), message: z.string() }).optional(),
  timing: z.object({
    dns: z.number(), connect: z.number(), tls: z.number(),
    request: z.number(), wait: z.number(), response: z.number(), total: z.number(),
  }),
  target: z.object({ url: z.string(), host: z.string(), port: z.number() }),
});
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| renderer → preload | Untrusted UI (could be compromised by XSS in a future feature). Only typed `window.api.*` exposed via contextBridge. |
| preload → main | Untrusted renderer data crossing via `ipcRenderer.invoke`. Every channel Zod-validated at the IPC handler. |
| main → filesystem | User-chosen data dir. Could be inside a cloud-synced folder. Mitigated by warning chip. Atomic writes prevent corruption. |
| main → JVM helper | Trusted main process writes JSON-RPC to helper stdin. Helper validates inputs from the JSON-RPC method. |
| main → target HTTP (outbound) | Untrusted user-configured URLs in the future. Phase 1 only ships a fixed probe URL. |
| helper → Spring project (future) | Read-only on user project paths. Validated in Task 2 via path prefix check (not exercised in Phase 1). |
